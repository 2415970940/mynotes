#

if [ ! -e $FILE ];then
  echo "$FILE no exist."
  exit 6
fi

if [ -f $FILE ];then
  echo "$FILE is common file."
elif [ -d $FILE ];then
  echo "$FILE is directory."
else
  echo "unknown"
fi
